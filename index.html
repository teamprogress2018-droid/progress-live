var mpView = 'week';           // 'week' | 'day'
var mpClientId = null;         // currently open client
var mpCurrentWeekStart = null; // Date — start of shown period
var mpSelectedWorkoutId = null;
var mpTargetDayStr = null;     // date string for "add workout to day"
var mpDisplaySettingsOpen = false;

// STORAGE: { [dateStr]: [{ workoutId, sections: [{name, type, duration, exercises:[{name,sets:[{kg,reps,rest,done}]}]}] }] }
window.MP_DATA = window.MP_DATA || {};

const MP_WEEK_DAYS_PL = ['Pon','Wt','Śr','Czw','Pt','Sob','Nie'];
const MP_MONTHS_PL = ['Sty','Lut','Mar','Kwi','Maj','Cze','Lip','Sie','Wrz','Paź','Lis','Gru'];

function mpGetMonday(d) {
  const dt = new Date(d);
  const day = dt.getDay();
  const diff = day === 0 ? -6 : 1 - day;
  dt.setDate(dt.getDate() + diff);
  dt.setHours(0,0,0,0);
  return dt;
}

function mpDateStr(d) { return d.toISOString().split('T')[0]; }

function mpFormatHeader(d) {
  const dayName = MP_WEEK_DAYS_PL[(d.getDay()+6)%7];
  const month = MP_MONTHS_PL[d.getMonth()];
  return `${dayName} ${d.getDate()} • ${month}`;
}

// ── OPEN MASTER PLANNER ──
function openMasterPlanner(clientId) {
  mpClientId = clientId || (CL.length ? CL[0].id : null);
  mpCurrentWeekStart = mpGetMonday(new Date());
  goTo('masterplanner');
  mpRender();
}

function closeMasterPlanner() { goTo('dashboard'); }

// ── MAIN RENDER ──
function mpRender() {
  mpRenderClientHeader();
  if (mpView === 'week') mpRenderWeekView();
  else mpRenderDayView();
  mpUpdateLastUpdated();
}

function mpUpdateLastUpdated() {
  const el = document.getElementById('mp-last-updated');
  if (el) el.textContent = 'Last updated: Just now';
}

function mpRenderClientHeader() {
  const header = document.getElementById('mp-client-header');
  const c = CL.find(x => x.id === mpClientId);
  if (!c || !header) return;
  header.style.display = 'flex';
  const av = document.getElementById('mp-client-avatar');
  const nm = document.getElementById('mp-client-name');
  const mt = document.getElementById('mp-client-meta');
  if (av) { av.textContent = getInit(c.name); }
  if (nm) nm.textContent = c.name;
  if (mt) mt.textContent = (c.goal || 'brak celu') + ' · ' + (c.level || '');
}

// ── WEEK VIEW ──
function mpRenderWeekView() {
  const container = document.getElementById('mp-weeks-container');
  if (!container) return;

  // Show 5 weeks starting from mpCurrentWeekStart
  const weeks = [];
  for (let w = 0; w < 5; w++) {
    const wStart = new Date(mpCurrentWeekStart);
    wStart.setDate(wStart.getDate() + w * 7);
    // Each week = one Thursday (or Monday) — Everfit shows by week column
    // Actually Everfit shows each WEEK as a column (each column = 1 week, showing Thursday label)
    // But the "Day by Day" view shows all 7 days
    // In "Week by Week": each column = one week, date = Monday of that week
    weeks.push(wStart);
  }

  container.innerHTML = weeks.map((wStart, wi) => {
    const isCurrentWeek = mpDateStr(wStart) === mpDateStr(mpGetMonday(new Date()));
    const workouts = mpGetWorkoutsForWeek(wStart);
    const hasWorkouts = workouts.length > 0;

    return `<div style="
      min-width:280px;flex:1;border-right:1px solid var(--border);
      display:flex;flex-direction:column;
      ${isCurrentWeek ? 'border-top:2px solid var(--accent);' : ''}
    ">
      <!-- Week header -->
      <div style="
        padding:8px 12px;background:var(--s2);border-bottom:1px solid var(--border);
        font-size:12px;font-weight:700;
        color:${isCurrentWeek ? 'var(--accent)' : 'var(--text)'};
        display:flex;align-items:center;justify-content:space-between;
      ">
        <span>${mpFormatWeekLabel(wStart)}</span>
        <button onclick="mpAddWorkoutToWeek('${mpDateStr(wStart)}')" style="
          background:none;border:none;color:var(--muted);font-size:18px;cursor:pointer;
          line-height:1;padding:0 2px;
        " title="Add workout">+</button>

function mpFormatWeekLabel(wStart) {
  // Everfit shows "May 21 • Thu" for the week
  const dayName = ['Sun','Mon','Tue','Wed','Thu','Fri','Sat'][wStart.getDay()];
  const months = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];
  return `${months[wStart.getMonth()]} ${wStart.getDate()} • ${dayName}`;
}

function mpGetWorkoutsForWeek(wStart) {
  const results = [];
  for (let d = 0; d < 7; d++) {
    const day = new Date(wStart);
    day.setDate(day.getDate() + d);
    const ds = mpDateStr(day);
    const dayWorkouts = (window.MP_DATA[ds] || []);
    dayWorkouts.forEach(w => results.push({ ...w, date: ds, dayIndex: d }));
  }
  return results;
}

function mpToggleSection(btn) {
  const body = btn.nextElementSibling;
  const arrow = btn.querySelector('span:first-child');
  if (body.style.display === 'none') {
    body.style.display = 'block';
    arrow.textContent = '▼';
  } else {
    body.style.display = 'none';
    arrow.textContent = '▶';
  }
}

function mpGetWorkoutData(wId) {

function mpBuildDefaultSections(w) {
  if (!w) return [];
  const exercises = (w.exercises || []).map(ex => ({
    name: typeof ex === 'string' ? ex : (ex.name || ex.n || 'Exercise'),
    thumb: '🏋️',
    type: 'weight',
    eachSide: false,
    tempo: '',
    sets: Array.from({length: parseInt(typeof ex === 'string' ? 3 : (ex.sets || 3))}, (_, i) => ({
      setNo: i + 1,
      kg: '',
      reps: typeof ex === 'string' ? '10' : (ex.reps || '10'),
      rest: '00:00',
      done: false
    }))
  }));

  // Split into logical sections
  if (exercises.length === 0) {
    return [{ name: 'Full Body Warm Up', type: 'amrap', duration: '10 min', exercises: [] }];
  }

  const third = Math.ceil(exercises.length / 3);
  const sections = [];
  const sectionNames = ['Full Body Cool Down', 'Legs', 'Full Body Warm Up'];
  const sectionTypes = ['regular', 'interval', 'amrap'];
  const sectionDurations = [null, '30min', '10 min'];

  for (let i = 0; i < Math.min(3, Math.ceil(exercises.length / third)); i++) {
    sections.push({
      name: sectionNames[i] || `Section ${i + 1}`,
      type: sectionTypes[i] || 'regular',
      duration: sectionDurations[i] || null,
      exercises: exercises.slice(i * third, (i + 1) * third)
    });
  }

  return sections.length ? sections : [{ name: 'Main', type: 'regular', exercises }];
}

// ── SET UPDATES ──
function mpUpdateSet(dateStr, exName, setIdx, field, value) {
  const dayData = window.MP_DATA[dateStr];
  if (!dayData) return;
  dayData.forEach(w => {
    (w.sections || []).forEach(sec => {
      (sec.exercises || []).forEach(ex => {
        if (ex.name === exName && ex.sets && ex.sets[setIdx]) {
          ex.sets[setIdx][field] = value;
        }
      });
    });
  });
}

function mpToggleDone(dateStr, exName, setIdx, btn) {
  const dayData = window.MP_DATA[dateStr];
  if (!dayData) return;
  let newDone = false;
  dayData.forEach(w => {
    (w.sections || []).forEach(sec => {
      (sec.exercises || []).forEach(ex => {
        if (ex.name === exName && ex.sets && ex.sets[setIdx]) {
          ex.sets[setIdx].done = !ex.sets[setIdx].done;
          newDone = ex.sets[setIdx].done;
        }
      });
    });
  });
  btn.style.background = newDone ? 'var(--teal)' : 'none';
  btn.style.borderColor = newDone ? 'var(--teal)' : 'var(--border2)';
  btn.textContent = newDone ? '✓' : '';
}

function mpAddSet(dateStr, exName) {
  const dayData = window.MP_DATA[dateStr];
  if (!dayData) return;
  dayData.forEach(w => {
    (w.sections || []).forEach(sec => {
      (sec.exercises || []).forEach(ex => {
        if (ex.name === exName) {
          const setNo = (ex.sets || []).length + 1;
          if (!ex.sets) ex.sets = [];
          ex.sets.push({ setNo, kg: '', reps: '10', rest: '00:00', done: false });
        }
      });
    });
  });
  mpRender();
}

function mpRemoveWorkout(dateStr, workoutId) {
  if (window.MP_DATA[dateStr]) {
    window.MP_DATA[dateStr] = window.MP_DATA[dateStr].filter(w => w.workoutId !== workoutId);
    mpRender();
  }
}

function mpDeleteExercise(dateStr, exName) {
  const dayData = window.MP_DATA[dateStr];
  if (!dayData) return;
  dayData.forEach(w => {
    (w.sections || []).forEach(sec => {
      sec.exercises = (sec.exercises || []).filter(ex => ex.name !== exName);
    });
  });
  mpRender();
  notify('Ćwiczenie usunięte');
}

function mpDuplicateExercise(dateStr, exName) {
  const dayData = window.MP_DATA[dateStr];
  if (!dayData) return;
  dayData.forEach(w => {
    (w.sections || []).forEach(sec => {
      const idx = (sec.exercises || []).findIndex(ex => ex.name === exName);
      if (idx >= 0) {
        const copy = JSON.parse(JSON.stringify(sec.exercises[idx]));
        copy.name += ' (kopia)';
        sec.exercises.splice(idx + 1, 0, copy);
      }
    });
  });
  mpRender();
  notify('Ćwiczenie zduplikowane');
}
